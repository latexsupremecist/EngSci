\section{Classes}

\begin{defn}[Class]
    We say a function $f:U\subseteq \R^n \mapsto \R^m$ is of class $n$ if its partial derivatives of order $n$ or less are continuous. We similarly define $C^\infty$ if all partial derivatives are continuous.
\end{defn}
